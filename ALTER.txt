iii>OR operator
Show the person who lives in Lake Town or Borat
SELECT Name FROM Student WHERE City ="Lake Town" OR City ="Borat";

iv>IN operator
Show the person who lives in Lake Town, or Borat, or Dum Dum
SELECT Name FROM Student WHERE City IN('Lake Town','Borat','Dum Dum');

iv>NOT IN operator
1>Show the person who does not live in Lake Town or Borat
SELECT Name FROM Student WHERE City NOT IN ('Lake Town','Borat');


v>Not equal operator
1>Show the person who NOT lives in Lake Town
SELECT Name FROM Student WHERE City!='Lake Town';
SELECT Name FROM Student WHERE City<> 'Lake Town';
